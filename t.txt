const axios = require('axios');
const fs = require('fs');
const path = require('path');
const { exec } = require('child_process');
const ffmpegPath = require('@ffmpeg-installer/ffmpeg').path;

module.exports = {
    name: 'tts3',
    aliases: ['boy'],
    category: 'tools',
    description: 'Convert text to voice (Boy)',
    usage: '.tts3 <text>',

    async execute(sock, msg, args, extra) {
        if (!args || args.length === 0) return await extra.reply("Please write something to convert.");
        const text = args.join(' ');
        if (text.length > 300) return await extra.reply("⚠️ Text is too long (maximum 300 characters allowed).");

        try {
            await sock.sendMessage(extra.from, { react: { text: '⏳', key: msg.key } });
            
            const response = await axios.post('https://tiktok-tts.weilnet.workers.dev/api/generation', {
                text: text,
                voice: 'en_us_006'
            }, { headers: { 'Content-Type': 'application/json' } });

            if (response.data.error || !response.data.data) throw new Error("API Error");

            const rawBuffer = Buffer.from(response.data.data, 'base64');
            const tempDir = require('os').tmpdir();
            const inputPath = path.join(tempDir, `tts3_in_${Date.now()}.mp3`);
            const outputPath = path.join(tempDir, `tts3_out_${Date.now()}.ogg`);

            fs.writeFileSync(inputPath, rawBuffer);
            const ffmpegCommand = `"${ffmpegPath}" -i "${inputPath}" -c:a libopus -b:a 128k -vbr on -compression_level 10 "${outputPath}"`;

            exec(ffmpegCommand, async (error) => {
                if (error) {
                    let errOpt = { audio: rawBuffer, mimetype: 'audio/mpeg', ptt: false, secureMetaServiceLabel: true };
                    if (!extra.isGroup) errOpt.ai = true;
                    return await sock.sendMessage(extra.from, errOpt, { quoted: msg });
                }
                try {
                    const finalBuffer = fs.readFileSync(outputPath);
                    await sock.sendMessage(extra.from, { react: { text: '✅', key: msg.key } });
                    let finalOpt = { audio: finalBuffer, mimetype: 'audio/ogg; codecs=opus', ptt: true, secureMetaServiceLabel: true };
                    if (!extra.isGroup) finalOpt.ai = true;
                    await sock.sendMessage(extra.from, finalOpt, { quoted: msg });
                } catch (e) {
                    await sock.sendMessage(extra.from, { react: { text: '❌', key: msg.key } });
                } finally {
                    if (fs.existsSync(inputPath)) fs.unlinkSync(inputPath);
                    if (fs.existsSync(outputPath)) fs.unlinkSync(outputPath);
                }
            });
        } catch (error) {
            await sock.sendMessage(extra.from, { react: { text: '❌', key: msg.key } });
            await extra.reply("❌ Failed to generate audio.");
        }
    }
};
